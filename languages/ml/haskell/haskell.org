- When to use (>>=) and when use do?
  1) if it seems like you're writing way *Too Many Lambdas* use ~do~

  3) Or use both

main :: IO ()
main = do
  pid <- getProcessID
  (filename:_) <- IO.getArgs
  bs <- BS.readFile filename
  let !maybeJson = decode bs :: Maybe Value
